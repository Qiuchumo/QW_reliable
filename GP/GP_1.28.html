with pm.Model() as model:
    _per = pm.HalfCauchy("_per", beta=2)
    _pdecay = pm.Gamma("_pdecay", alpha=10, beta=0.075)
    period  = pm.Normal("period", mu=2, sd=0.05) # 2
    _psmooth = pm.Gamma("_psmooth ", alpha=4, beta=3)
    cov_seasonal = _per * pm.gp.cov.Periodic(1, period, _psmooth) \
                            * pm.gp.cov.Matern52(1, _pdecay)
    gp_seasonal = pm.gp.Marginal(cov_func=cov_seasonal)

    # long term trend
    _trend = pm.HalfCauchy("_trend", beta=1)
    _trend = pm.Gamma("_trend", alpha=4, beta=0.1)
    cov_trend = _trend**2 * pm.gp.cov.ExpQuad(1, _trend)
    gp_trend = pm.gp.Marginal(cov_func=cov_trend) 
    
    gp = gp_seasonal + gp_trend
